import Towers.YM.LatticeGauge
import Mathlib.Analysis.Complex.Basic

namespace TheoremaAureum.Towers.YM

open Matrix

def plaquette (d L : ℕ) (U : GaugeConfig d L) (x : Lattice d L) (μ ν : Fin d) 
    (hμν : μ ≠ ν) : G := 
  U (x, μ) * U (x + Pi.single μ 1, ν) * (U (x + Pi.single ν 1, μ))⁻¹ * (U (x, ν))⁻¹

noncomputable def wilsonAction (d L : ℕ) (β : ℝ) (U : GaugeConfig d L) : ℝ := 
  β * ∑ x : Lattice d L, ∑ μ : Fin d, ∑ ν : Fin d, 
    if h : μ ≠ ν then 1 - (1/2) * (plaquette d L U x μ ν h).trace.re else 0

end TheoremaAureum.Towers.YM
